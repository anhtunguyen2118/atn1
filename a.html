Phản hồi:") # logging.debug(response_text[:1000]) # Log một phần phản hồi để debug return None except requests.exceptions.RequestException as e_req: logging.error(f"Lỗi Request khi giải CAPTCHA: {e_req}") except Exception as e: logging.error(f"Lỗi không xác định khi giải CAPTCHA: {e}", exc_info=True) return None def initialize_driver(profile_dir_path, profile_name_for_log="N/A", win_x=None, win_y=None, width=None, height=None): # ...
